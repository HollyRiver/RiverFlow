# step 1
X = pd.get_dummies(df[['toeic', 'gpa']])
y = df.employment

# step 2
predictr = sklearn.linear_model.LogisticRegression()

# step 3
predictr.fit(X, y)

# step 4
df = df.assign(employment_hat = predictr.predict(X))
